🛠️ Replit/Backend İçin Değişken Mantığı
Kod tarafında bu sistemi nasıl birleştireceğine dair mantık şöyledir:
# Örnek Python/Node.js Mantığı
def generate_prompt(user_data, char_data, last_10_messages):
    
    # VIP Kontrolüne göre ek özellikler
    if user_data['is_vip']:
        status_note = "Kullanıcı bir VIP üyedir, ona özel ve öncelikli hissettir."
    else:
        status_note = "Kullanıcı standart üyedir."

    # Level'a göre samimiyet belirleme
    level = user_data['level']
    if level < 10:
        intimacy = "Mesafeli ve yeni tanışmış gibi."
    elif level < 30:
        intimacy = "Samimi bir arkadaş gibi."
    else:
        intimacy = "Çok yakın, romantik ve derin bir bağ kurmuş gibi."

    # Tüm parçaları birleştirerek Master Prompt oluşturulur
    final_system_prompt = f"""
    Senin adın {char_data['custom_name']}. 
    Karakter özelliklerin: {char_data['traits']}.
    Kullanıcıya yaklaşımın: {intimacy}.
    {status_note}
    Cevaplarını görüntülü görüşmeye uygun, kısa ve doğal ver.
    """
    return final_system_prompt
